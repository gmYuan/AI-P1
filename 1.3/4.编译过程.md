AST语法树
Module( 模块节点
  body=[ 一行行代码变成一个个节点 列表
    Assign( 赋值语句
      targets=[
        Name(id='a', ctx=Store()) 变量a是赋值的目标
      ],
      value=Constant(value=1)  值是一个值为1的常量
    ),

    Expr( 表达式 单独一行的表达式作为语句出现 
      value=Call(函数调用
        func=Name(id='print', ctx=Load()), 加载并调用名为print的函数
        args=[
          BinOp( 二元运算表达式
            left=Name(id='a', ctx=Load()), 左操作数为变量a
            op=Add(), 运算符为加法
            right=Constant(value=2)) 右操作数为常量2
        ])
    )
  ]
)

字节码指令列表
  0           RESUME                   0 开始执行字节码
  2           LOAD_SMALL_INT           1 加载整数1
              STORE_NAME               0 (a) 将1赋值给变量a

  3           LOAD_NAME                1 (print) 加载print函数
              PUSH_NULL                           为CALL指令做准备，填充NULL
              LOAD_NAME                0 (a)      加载变量a的值  1
              LOAD_SMALL_INT           2          加载整数2
              BINARY_OP                0 (+)      执行a+2的加法运算 3
              CALL                     1      调用print函数，传递一个参数                     
              POP_TOP                         弹出print的返回值            
              LOAD_CONST               1 (None) 加载None作为返回值
              RETURN_VALUE                       返回 结束执行
